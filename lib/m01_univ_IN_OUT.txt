 	; universal output ( works for all registers )
   	.macro    OUTU
   	.if	@0 < 0x40
      	OUT	@0,@1         
	.else
      	STS	@0,@1
   	.endif
   	.endm
	
	
 	; universal input ( works for all registers )
   	.macro    INU
   	.if	@0 < 0x40
      	IN	@0,@1         
	.else
      	LDS	@0,@1
   	.endif
   	.endm
